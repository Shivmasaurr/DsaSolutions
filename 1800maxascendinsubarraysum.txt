class Solution {
public:
    int maxAscendingSum(vector<int>& nums) {
        int max_sum = nums[0], local_sum = nums[0];
        for (int i = 1; i < nums.size(); i++) {
            if (nums[i] > nums[i - 1])
                local_sum += nums[i];
            else
                local_sum = nums[i];
            max_sum = max(max_sum, local_sum);
        }
        return max_sum;
    }
};
